FUNCTION TapCard (Balance, Cost)
    IF(Balance>=Cost)
        STORE "Balance" WITH Balance-Cost
        GatewayOpen()
    ELSE
        IF(Topup)
            READ AND WRITE "Balance" WITH NewBalance
            TapCard(Balance)
        ELSE
            GatewayClose()
